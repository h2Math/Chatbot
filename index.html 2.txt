        function addLesson() { 
            const sid=document.getElementById('lessonStudent').value; 
            let start=document.getElementById('lessonStart').value; 
            let end=document.getElementById('lessonEnd').value; 
            const topic=document.getElementById('lessonTopic').value; 
            const notes=document.getElementById('lessonNotes').value; 
            if(!sid||!start||!end){ showMessage('Select student and times','error','lessonFormMessage'); return; } 
            
            // Parse local times
            const startLocal = new Date(start + ':00');
            const endLocal = new Date(end + ':00');
            
            if(startLocal >= endLocal){ showMessage('End time must be after start','error','lessonFormMessage'); return; } 
            const student=students.find(s=>s.id===sid); 
            if(!student){ showMessage('Student not found','error','lessonFormMessage'); return; } 
            
            // Store as simple format: "YYYY-MM-DD HH:MM:SS"
            const startStr = start.replace('T', ' ') + ':00';
            const endStr = end.replace('T', ' ') + ':00';
            
            const hours = (endLocal - startLocal) / 3600000; 
            const earnings = student.rateType === 'hourly' ? hours * student.rateAmount : student.rateAmount;
            
            lessons.push({
                id:Date.now().toString(),
                studentId:sid,
                studentName:student.name,
                groupId:student.groupId,
                start:startStr,
                end:endStr,
                topic:topic,
                notes:notes,
                earnings:earnings,
                hours:hours
            }); 
            saveLessons(); 
            renderCalendar(); 
            updateLessonStats(); 
            updateBackupStats(); 
            document.getElementById('lessonTopic').value=''; 
            document.getElementById('lessonNotes').value=''; 
            showMessage(`✅ Lesson added for ${student.name} - ${hours.toFixed(1)} hrs = $${earnings.toFixed(2)}`,'success','lessonFormMessage'); 
            const now=new Date(); 
            now.setHours(now.getHours()+1); 
            now.setMinutes(0); 
            document.getElementById('lessonStart').value=getDateTimeLocal(now); 
            document.getElementById('lessonStart').dispatchEvent(new Event('change')); 
        }

        // FIXED: Calendar rendering using local date comparison
        function renderCalendar() {
            const container = document.getElementById('calendarViewContainer');
            if (!container) return;
            
            const firstDayOfMonth = new Date(currentCalendarYear, currentCalendarMonth, 1);
            const lastDayOfMonth = new Date(currentCalendarYear, currentCalendarMonth + 1, 0);
            const startDay = firstDayOfMonth.getDay();
            const daysInMonth = lastDayOfMonth.getDate();
            
            const monthNames = ['January', 'February', 'March', 'April', 'May', 'June', 'July', 'August', 'September', 'October', 'November', 'December'];
            
            let calendarHTML = `
                <div class="calendar-container">
                    <div class="calendar-header-nav">
                        <button class="calendar-nav-btn" onclick="changeMonth(-1)">◀ Previous</button>
                        <div class="calendar-title">${monthNames[currentCalendarMonth]} ${currentCalendarYear}</div>
                        <button class="calendar-nav-btn" onclick="changeMonth(1)">Next ▶</button>
                    </div>
                    <div class="calendar-weekdays">
                        <div class="calendar-weekday">Sun</div><div class="calendar-weekday">Mon</div><div class="calendar-weekday">Tue</div>
                        <div class="calendar-weekday">Wed</div><div class="calendar-weekday">Thu</div><div class="calendar-weekday">Fri</div>
                        <div class="calendar-weekday">Sat</div>
                    </div>
                    <div class="calendar-days">
            `;
            
            for (let i = 0; i < startDay; i++) {
                calendarHTML += `<div class="calendar-day empty"></div>`;
            }
            
            for (let day = 1; day <= daysInMonth; day++) {
                // Find lessons for this specific day using local date comparison
                const dayLessons = lessons.filter(l => {
                    const lessonDate = parseLocalDateTime(l.start);
                    return lessonDate.getFullYear() === currentCalendarYear && 
                           lessonDate.getMonth() === currentCalendarMonth && 
                           lessonDate.getDate() === day;
                });
                
                // Sort lessons by time
                dayLessons.sort((a, b) => {
                    const dateA = parseLocalDateTime(a.start);
                    const dateB = parseLocalDateTime(b.start);
                    return dateA - dateB;
                });
                
                calendarHTML += `<div class="calendar-day" data-day="${day}">
                    <div class="day-number">${day}</div>`;
                
                for (const lesson of dayLessons) {
                    const lessonDate = parseLocalDateTime(lesson.start);
                    const timeStr = lessonDate.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' });
                    
                    calendarHTML += `
                        <div class="calendar-lesson" onclick="editLessonFromCalendar('${lesson.id}', event)">
                            <span class="lesson-time">${timeStr}</span>
                            <span class="lesson-student"> ${escapeHtml(lesson.studentName)}</span>
                            <div style="font-size:0.6rem; color:#8892b0;">💰 $${lesson.earnings.toFixed(2)}</div>
                        </div>
                    `;
                }
                
                calendarHTML += `</div>`;
            }
            
            const totalCells = startDay + daysInMonth;
            const remainingCells = 42 - totalCells;
            for (let i = 0; i < remainingCells; i++) {
                calendarHTML += `<div class="calendar-day empty"></div>`;
            }
            
            calendarHTML += `
                    </div>
                    <div class="calendar-legend">
                        <div class="legend-item"><div class="legend-color"></div><span>Lesson scheduled (click to edit notes)</span></div>
                    </div>
                </div>
            `;
            
            container.innerHTML = calendarHTML;
        }
        
        function changeMonth(delta) {
            let newMonth = currentCalendarMonth + delta;
            let newYear = currentCalendarYear;
            if (newMonth < 0) { newMonth = 11; newYear--; }
            if (newMonth > 11) { newMonth = 0; newYear++; }
            currentCalendarMonth = newMonth;
            currentCalendarYear = newYear;
            renderCalendar();
        }

        // Export to Calendar function
        function escapeICS(str) { 
            if (!str) return ''; 
            return str.replace(/[\\;,\n]/g, function(match) {
                if (match === '\\') return '\\\\';
                if (match === ';') return '\\;';
                if (match === ',') return '\\,';
                if (match === '\n') return '\\n';
                return match;
            });
        }

        async function shareInvoice() {
            if (!currentInvoiceElement) { showMessage('Generate invoice first', 'error', 'invoiceFormMessage'); return; }
            showMessage('Generating PDF...', 'success', 'invoiceFormMessage');
            try {
                const groupId = document.getElementById('invoiceGroup').value;
                const parent = groups.find(g => g.id === groupId);
                const fromDate = document.getElementById('invoiceFromDate').value;
                const toDate = document.getElementById('invoiceToDate').value;
                const totalElement = document.querySelector('#invoiceContainer .invoice-total p:last-child');
                let totalAmount = '0.00';
                if (totalElement) {
                    const match = totalElement.innerText.match(/\$([\d.]+)/);
                    if (match) totalAmount = match[1];
                }
                const formatDM = (dateStr) => {
                    const d = new Date(dateStr);
                    const day = String(d.getDate()).padStart(2, '0');
                    const month = String(d.getMonth() + 1).padStart(2, '0');
                    return `${day}/${month}`;
                };
                const dateRange = `${formatDM(fromDate)}-${formatDM(toDate)}`;
                const messageText = `Hi ${parent?.name || 'Parent'}, $${totalAmount} for ${dateRange}. Thank you.`;
                const filename = `Invoice.pdf`;
                const el = currentInvoiceElement.cloneNode(true);
                el.style.width = '800px';
                el.style.margin = '0';
                el.style.padding = '30px';
                el.style.backgroundColor = 'white';
                el.style.position = 'absolute';
                el.style.left = '-9999px';
                document.body.appendChild(el);
                const canvas = await html2canvas(el, { scale: 2, backgroundColor: '#fff' });
                document.body.removeChild(el);
                const { jsPDF } = window.jspdf;
                const img = canvas.toDataURL('image/png');
                const w = 210;
                const h = (canvas.height * w) / canvas.width;
                const pdf = new jsPDF('p', 'mm', 'a4');
                pdf.addImage(img, 'PNG', 0, 0, w, h);
                const blob = pdf.output('blob');
                const file = new File([blob], filename, { type: 'application/pdf' });
                if (navigator.share && navigator.canShare && navigator.canShare({ files: [file] })) {
                    await navigator.share({
                        title: 'Invoice',
                        text: messageText,
                        files: [file]
                    });
                    showMessage('Invoice shared!', 'success', 'invoiceFormMessage');
                } else {
                    pdf.save(filename);
                    showMessage('PDF downloaded!', 'success', 'invoiceFormMessage');
                }
            } catch(e) {
                console.error(e);
                showMessage('Error generating PDF', 'error', 'invoiceFormMessage');
            }
        }
